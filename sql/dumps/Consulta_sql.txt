DESCRIBE PARCEIRO.CLIENT_PRODUCT;
SELECT * FROM PARCEIRO.LOJA;
SELECT * FROM PARCEIRO.PRODUTO;
SELECT * FROM USUARIOS.CLIENTES;
SELECT * FROM PARCEIRO.CLIENT_PRODUCT;

SELECT NOME, ID_USER, CPF FROM USUARIOS.CLIENTES WHERE NASCIMENTO <> '2001-01-21' ORDER BY NOME ;

SELECT CPF, NOME , ID_USER, PRODUTO, VALOR, QUANTIDADE
FROM USUARIOS.CLIENTES AS UC
INNER JOIN CLIENT_PRODUCT AS CP 
ON UC.ID_USER = CP.USU_ARIO
INNER JOIN PARCEIRO.PRODUTO AS PP
ON CP.VEN_DA = PP.ID_VENDA ORDER BY NOME;
